496 [ExecuteInEditMode]
497 private void Awake()
498 {
499 // If Material Parameters is in old format we need to parse old parameters and come up with equivalent new
500 // ones
501#if UNITY_EDITOR
502 bool updated = false;
503#endif
504
505 if (ObjectVersion == 1)
506 {
507 Roughness = 1 - SmoothnessOld;
508
509 ObjectVersion = 2;
510#if UNITY_EDITOR
511 updated = true;
512#endif
513 }
514
515 if (ObjectVersion == 2)
516 {
517 Solver.ZibraLiquid instance = GetComponent<Solver.ZibraLiquid>();
518
519 // if not a newly created liquid instance
520 //(material parameters are created before liquid)
521 if (instance != null)
522 {
523 const float TotalScale = 0.33f;
524 float SimulationScale =
525 TotalScale * (instance.ContainerSize.x + instance.ContainerSize.y + instance.ContainerSize.z);
526
527 ScatteringAmount *= SimulationScale;
528 AbsorptionAmount *= SimulationScale;
529 }
530
531 ObjectVersion = 3;
532
533#if UNITY_EDITOR
534 updated = true;
535#endif
536 }
537
538#if UNITY_EDITOR
539 if (updated)
540 {
541 // Can't mark object dirty in Awake, since scene is not fully loaded yet
542 UnityEditor.SceneManagement.EditorSceneManager.sceneOpened += OnSceneOpened;
543 }
544#endif
545 }
