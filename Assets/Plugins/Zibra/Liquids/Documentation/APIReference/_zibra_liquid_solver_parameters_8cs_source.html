422 internal void ValidateParameters()
423 {
424#if ZIBRA_LIQUID_PRO_VERSION
425 if (Application.isPlaying)
426 {
427 if (MaximumSpeciesCount == 0)
428 {
429 MaximumSpeciesCount = AdditionalParticleSpecies.Count + MAX_RUNTIME_ADDED_SPECIES;
430 }
431 else
432 {
433 if (AdditionalParticleSpecies.Count >= MaximumSpeciesCount)
434 {
435 int toDelete = AdditionalParticleSpecies.Count - MaximumSpeciesCount + 1;
436 AdditionalParticleSpecies.RemoveRange(MaximumSpeciesCount - 1, toDelete);
437 Debug.LogWarning("Zibra Liquids: Exceeded maximum number of species in current instance");
438 }
439 }
440 }
441
442 foreach (var param in AdditionalParticleSpecies)
443 {
444 int ZeroParameterCount = 0;
445 if (param.Gravity == Vector3.zero)
446 ZeroParameterCount++;
447 ValidateGravity(ref param.Gravity);
448 if (param.FluidStiffness == 0.0f)
449 {
450 ZeroParameterCount++;
451 param.FluidStiffness = DEFAULT_STIFFNESS;
452 }
453 if (param.ParticleDensity == 0.0f)
454 {
455 ZeroParameterCount++;
456 param.ParticleDensity = DEFAULT_DENSITY;
457 }
458 if (param.Viscosity == 0.0f)
459 ZeroParameterCount++;
460 if (param.MaximumVelocity == 0.0f)
461 ZeroParameterCount++;
462
463 if (ZeroParameterCount == 5) // if parameters are initialized as 0
464 {
465 param.MaximumVelocity = DEFAULT_MAX_VELOCITY;
466 param.Viscosity = DEFAULT_VISCOSITY;
467 param.Gravity = new Vector3(0.0f, DEFAULT_GRAVITY, 0.0f);
468 }
469 }
470#endif
471
472 ValidateGravity(ref Gravity);
473 }
