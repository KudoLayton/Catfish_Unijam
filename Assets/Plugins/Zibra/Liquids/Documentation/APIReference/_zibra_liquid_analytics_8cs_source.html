34 public static void TrackLiquidInitialization(ZibraLiquid liquid)
35 {
36 EditorPrefs.SetBool("ZibraLiquidsTracking_ZibraLiquidsUsed", true);
37
38#if ZIBRA_LIQUID_PAID_VERSION
39 if (liquid.InitialState == ZibraLiquid.InitialStateType.BakedLiquidState && liquid.BakedInitialStateAsset != null)
40 {
41 EditorPrefs.SetBool("ZibraLiquidsTracking_BakedStateUsed", true);
42 }
43#endif
44 var currentRendererMode = liquid.CurrentRenderingMode;
45 switch (currentRendererMode)
46 {
47 case ZibraLiquid.RenderingMode.MeshRender:
48 EditorPrefs.SetBool("ZibraLiquidsTracking_MashRendererUsed", true);
49 break;
50 case ZibraLiquid.RenderingMode.UnityRender:
51 EditorPrefs.SetBool("ZibraLiquidsTracking_UnityRendererUsed", true);
52 break;
53 }
54
55 if (liquid.EnableDownscale)
56 {
57 EditorPrefs.SetBool("ZibraLiquidsTracking_RenderDownscaleUsed", true);
58 }
59
60 int currentNodeCount = liquid.GridNodeCount;
61 int previousMaxNodeCount = EditorPrefs.GetInt("ZibraLiquidsTracking_MaxNodeCount", 0);
62 int previousMinNodeCount = EditorPrefs.GetInt("ZibraLiquidsTracking_MinNodeCount", int.MaxValue);
63 EditorPrefs.SetInt("ZibraLiquidsTracking_MaxNodeCount", Math.Max(currentNodeCount, previousMaxNodeCount));
64 EditorPrefs.SetInt("ZibraLiquidsTracking_MinNodeCount", Math.Min(currentNodeCount, previousMinNodeCount));
65
66 int currentMaxNumParticles = liquid.MaxNumParticles;
67 int previousMaxMaxNumParticles = EditorPrefs.GetInt("ZibraLiquidsTracking_MaxMaxParticleCount", 0);
68 int previousMinMaxNumParticles = EditorPrefs.GetInt("ZibraLiquidsTracking_MinMaxParticleCount", int.MaxValue);
69 EditorPrefs.SetInt("ZibraLiquidsTracking_MaxMaxParticleCount", Math.Max(currentMaxNumParticles, previousMaxMaxNumParticles));
70 EditorPrefs.SetInt("ZibraLiquidsTracking_MinMaxParticleCount", Math.Min(currentMaxNumParticles, previousMinMaxNumParticles));
71
72 int currentGridResolution = liquid.GridResolution;
73 int previousMaxGridResolution = EditorPrefs.GetInt("ZibraLiquidsTracking_MaxGridResolution", 0);
74 int previousMinGridResolution = EditorPrefs.GetInt("ZibraLiquidsTracking_MinGridResolution", int.MaxValue);
75 EditorPrefs.SetInt("ZibraLiquidsTracking_MaxGridResolution", Math.Max(currentGridResolution, previousMaxGridResolution));
76 EditorPrefs.SetInt("ZibraLiquidsTracking_MinGridResolution", Math.Min(currentGridResolution, previousMinGridResolution));
77
78 foreach (var item in liquid.GetManipulatorList())
79 {
80 var manipulatorType = item.GetManipulatorType();
81 var sdf = item.GetComponent<SDFObject>();
82
83 if (sdf == null)
84 continue;
85
86 switch (manipulatorType)
87 {
88 case Manipulator.ManipulatorType.Emitter:
89 case Manipulator.ManipulatorType.Void:
90 case Manipulator.ManipulatorType.ForceField:
91 case Manipulator.ManipulatorType.Detector:
92 case Manipulator.ManipulatorType.SpeciesModifier:
93 TrackManipulatorSDF(sdf, manipulatorType.ToString());
94 break;
95 }
96 }
97
98 foreach (var item in liquid.GetColliderList())
99 {
100 var sdf = item.GetComponent<SDFObject>();
101
102 if (sdf == null)
103 continue;
104
105 TrackManipulatorSDF(sdf, "Collider");
106 }
107 }

109 public static void CheckSendPeriod()
110 {
111 int dateNow = GetCurrentTimeAsUnixTimestamp();
112
113 if (!EditorPrefs.HasKey("ZibraLiquidsTracking_LastSentDate"))
114 {
115 EditorPrefs.SetInt("ZibraLiquidsTracking_LastSentDate", dateNow);
116 return;
117 }
118
119 int lastSentDate = EditorPrefs.GetInt("ZibraLiquidsTracking_LastSentDate");
120
121 if (dateNow - lastSentDate > REPORT_PERIOD)
122 {
123 ZibraLiquidAnalyticsStruct data = GetAnalyticsData();
124 string jsonData = JsonUtility.ToJson(data);
125 ZibraLiquidAnalyticsSender.SendAnalyticsData(jsonData);
126 }

388 private static UnityWebRequestAsyncOperation request;
389
390 public static void SendAnalyticsData(string jsonData)
391 {
392 if (request != null)
393 {
394 return;
395 }
396
397 request = UnityWebRequest.Post(ANALYTIC_API_URL, jsonData).SendWebRequest();
398 request.completed += UpdateKeyRequest;
399 }
400
401 private static void UpdateKeyRequest(AsyncOperation obj)
402 {
403 if (request == null || !request.isDone)
404 {
405 return;
406 }
407
408#if UNITY_2020_2_OR_NEWER
409 if (request.webRequest.result != UnityWebRequest.Result.Success)
410#else
411 if (request.webRequest.isHttpError || request.webRequest.isNetworkError)
412#endif
413 {
414 request.webRequest.Dispose();
415 request = null;
416 return;
417 }
418
419 if (request.webRequest.responseCode != 201)
420 {
421 request.webRequest.Dispose();
422 request = null;
423 return;
424 }
425
426 ZibraLiquidAnalyticsData.CleanAnalyticData();
427 request.webRequest.Dispose();
428 request = null;
429 }
430 }
431}
