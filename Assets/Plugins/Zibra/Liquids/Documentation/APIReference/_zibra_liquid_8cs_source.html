1380 }
1381#endif

1383#if ZIBRA_LIQUID_PAID_VERSION
1384 if (InitialState == ZibraLiquid.InitialStateType.NoParticles || BakedInitialStateAsset == null)

1399#if ZIBRA_LIQUID_PAID_VERSION

1403#endif

1408

1422#if ZIBRA_LIQUID_FREE_VERSION

1428 switch (SystemInfo.graphicsDeviceType)
1429 {
1430 case GraphicsDeviceType.Direct3D11:

1432 case GraphicsDeviceType.Switch:

1435 case GraphicsDeviceType.XboxOneD3D12:

1441 break;

1445 Marshal.AllocHGlobal(Marshal.SizeOf(initializeGPUReadbackParamsBridgeParams));
1446 Marshal.StructureToPtr(initializeGPUReadbackParamsBridgeParams, nativeCreateInstanceBridgeParams, true);
1447
1448 SolverCommandBuffer.Clear();

1450 ZibraLiquidBridge.EventID.InitializeGpuReadback,
1451 nativeCreateInstanceBridgeParams);
